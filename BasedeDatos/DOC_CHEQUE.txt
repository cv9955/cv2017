CREATE TABLE  "DOC_CHEQUES" 
   (	
    "ID" NUMBER NOT NULL ENABLE, --> DOC_SEQ
    "DOC_TIPO_ID" NUMBER   --> 10 PROPIOS 11 CLIENTE 12 OTROS
	"KEY" NUMBER,  --> MAX(KEY) + 1 WHERE DOC_TIPO_ID 
    "FECHA" DATE, 
	"VENCIMIENTO" DATE, 	
	"CHEQUE_NUM" NUMBER, 
	"BANCO_NUM" VARCHAR2(20),
	"CUENTA_NUM" VARCHAR2(20), 	
	"RSOCIAL" VARCHAR2(200), 

	"BANCO_ID" NUMBER,	
	"CLI_DFISCAL_ID" NUMBER, 

	"IMPORTE" NUMBER(*,2), 
	"ORIGEN" NUMBER,  -- 1 RECIBO -- 2 RECIBO2 -- 10 CUENTA VICTORIA
	"ORIGEN_REF" NUMBER, -- CAB_RECIBO_ID / DOC_CUENTA_ID
	"SALIDA" NUMBER,  -- 1 OPAGA  -- 2 OPAGO2  -- 10 CUENTA VICTORIA
	"SALIDA_REF" NUMBER, -- PROV_PAGO_ID / DOC_CUENTA_ID
 )
 
 CREATE TABLE  "DOCS_PROPIOS" 
   (	"ID" NUMBER NOT NULL ENABLE, 
	"FECHA" DATE, 
	"CHEQUE" NUMBER, 
	"VENCIMIENTO" DATE, 
	"IMPORTE" NUMBER(*,2), 
	"KEY" NUMBER, 
	"SALIDA" NUMBER, 
	"SALIDA_REF" NUMBER, 
	"CHEQUERA" NUMBER, 
	"ID_BANCO" NUMBER, 
	"CUENTA_ID" NUMBER, 
	"FECHA_DEPOSITO" DATE, 
	 CONSTRAINT "DOCS_PROPIOS_PK" PRIMARY KEY ("ID") ENABLE
   )
   
SELECT ID,FECHA,KEY,ORIGEN + 10 DOC_TIPO,
	CHEQUE CHEQUE_NUM,BANCO BANCO_NUM,CUENTA CUENTA_NUM,RSOCIAL,DOCS_BANCOS_ID BANCO_ID,VENCIMIENTO,IMPORTE,
ORIGEN,ORIGEN_REF,SALIDA, SALIDA_REF
FROM DOCS_CHEQUES
UNION ALL
SELECT ID,FECHA,KEY,10 DOC_TIPO,
CHEQUE CHEQUE_NUM,NULL BANCO_NUM,NULL CUENTA_NUM,NULL RSOCIAL,NULL BANCO_ID,VENCIMIENTO,IMPORTE,
10 ORIGEN ,CUENTA_ID ORIGEN_REF,
SALIDA, SALIDA_REF
FROM DOCS_PROPIOS
